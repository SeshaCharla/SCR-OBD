\subsection{Logarithm Formulae}
\begin{align*}
    \log xy &= \log x + \log y\\
    \frac{d}{dt} \log x &= \frac{\dot x}{x}
\end{align*}
Let, $L_x = \log x$
\begin{align*}
    \implies \log \dot L_x &= \log \dot x - \log x\\
    \implies \log \dot x &= \log \dot L_x  + L_x
\end{align*}
